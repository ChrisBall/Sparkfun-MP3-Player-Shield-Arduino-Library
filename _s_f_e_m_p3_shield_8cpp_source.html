173 uint8_t SFEMP3Shield::vs_init() {
174 
175  //Initialize VS1053 chip
176 
177  //Reset if not already
178  delay(100); // keep clear of anything prior
179  digitalWrite(MP3_RESET, LOW); //Shut down VS1053
180  delay(100);
181 
182  //Bring out of reset
183  digitalWrite(MP3_RESET, HIGH); //Bring up VS1053
184 
185  //From section 7.6 of datasheet, max SCI reads are CLKI/7.
186  //Assuming CLKI = 12.288MgHz for Shield and 16.0MgHz for Arduino
187  //The VS1053's internal clock multiplier SCI_CLOCKF:SC_MULT is 1.0x after power up.
188  //For a maximum SPI rate of 1.8MgHz = (CLKI/7) = (12.288/7) the VS1053's default.
189 
190  //Warning:
191  //Note that spi transfers interleave between SdCard and VS10xx.
192  //Where Sd2Card.cpp sets SPCR & SPSR each and every transfer
193 
194  //The SDfatlib using SPI_FULL_SPEED results in an 8MHz spi clock rate,
195  //faster than initial allowed spi rate of 1.8MgHz.
196 
197  // set initial mp3's spi to safe rate
198  spiRate = SPI_CLOCK_DIV16; // initial contact with VS10xx at slow rate
199  delay(10);
200 
201  //Let's check the status of the VS1053
202  int MP3Mode = Mp3ReadRegister(SCI_MODE);
203 
204 /*
205  Serial.print(F("SCI_Mode (0x4800) = 0x"));
206  Serial.println(MP3Mode, HEX);
207 
208  int MP3Status = Mp3ReadRegister(SCI_Status);
209  Serial.print(F("SCI_Status (0x48) = 0x"));
210  Serial.println(MP3Status, HEX);
211 
212  int MP3Clock = Mp3ReadRegister(SCI_CLOCKF);
213  Serial.print(F("SCI_ClockF = 0x"));
214  Serial.println(MP3Clock, HEX);
215  */
216 
217  if(MP3Mode != (SM_LINE1 | SM_SDINEW)) return 4;
218 
219  //Now that we have the VS1053 up and running, increase the internal clock multiplier and up our SPI rate
220  Mp3WriteRegister(SCI_CLOCKF, 0x6000); //Set multiplier to 3.0x
221  //Internal clock multiplier is now 3x.
222  //Therefore, max SPI speed is 52MgHz.
223  spiRate = SPI_CLOCK_DIV4; //use safe SPI rate of (16MHz / 4 = 4MHz)
224  delay(10); // settle time
225 
226  //test reading after data rate change
227  int MP3Clock = Mp3ReadRegister(SCI_CLOCKF);
228  if(MP3Clock != 0x6000) return 5;
229 
230  setVolume(40, 40);
231  // one would think the following patch would over write the volume.
232  // But the SCI_VOL register space is not in the VSdsp's WRAM space.
233  // Note to keep an eye on it for future patches.
234 
235  if(VSLoadUserCode("patches.053")) return 6;
236 
237  delay(100); // just a good idea to let settle.
238 
239  return 0; // indicating all was good.
240 }
