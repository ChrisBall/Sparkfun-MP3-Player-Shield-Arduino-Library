157 uint8_t SFEMP3Shield::vs_init() {
158 
159  //Initialize VS1053 chip
160 
161  //Reset if not already
162  delay(100); // keep clear of anything prior
163  digitalWrite(MP3_RESET, LOW); //Shut down VS1053
164  delay(100);
165 
166  //Bring out of reset
167  digitalWrite(MP3_RESET, HIGH); //Bring up VS1053
168 
169  //From section 7.6 of datasheet, max SCI reads are CLKI/7.
170  //Assuming CLKI = 12.288MgHz for Shield and 16.0MgHz for Arduino
171  //The VS1053's internal clock multiplier SCI_CLOCKF:SC_MULT is 1.0x after power up.
172  //For a maximum SPI rate of 1.8MgHz = (CLKI/7) = (12.288/7) the VS1053's default.
173 
174  //Warning:
175  //Note that spi transfers interleave between SdCard and VS10xx.
176  //Where Sd2Card.cpp sets SPCR & SPSR each and every transfer
177 
178  //The SDfatlib using SPI_FULL_SPEED results in an 8MHz spi clock rate,
179  //faster than initial allowed spi rate of 1.8MgHz.
180 
181  // set initial mp3's spi to safe rate
182  spiRate = SPI_CLOCK_DIV16; // initial contact with VS10xx at slow rate
183  delay(10);
184 
185  //Let's check the status of the VS1053
186  int MP3Mode = Mp3ReadRegister(SCI_MODE);
187 
188 /*
189  Serial.print(F("SCI_Mode (0x4800) = 0x"));
190  Serial.println(MP3Mode, HEX);
191 
192  int MP3Status = Mp3ReadRegister(SCI_Status);
193  Serial.print(F("SCI_Status (0x48) = 0x"));
194  Serial.println(MP3Status, HEX);
195 
196  int MP3Clock = Mp3ReadRegister(SCI_CLOCKF);
197  Serial.print(F("SCI_ClockF = 0x"));
198  Serial.println(MP3Clock, HEX);
199  */
200 
201  if(MP3Mode != (SM_LINE1 | SM_SDINEW)) return 4;
202 
203  //Now that we have the VS1053 up and running, increase the internal clock multiplier and up our SPI rate
204  Mp3WriteRegister(SCI_CLOCKF, 0x6000); //Set multiplier to 3.0x
205  //Internal clock multiplier is now 3x.
206  //Therefore, max SPI speed is 52MgHz.
207  spiRate = SPI_CLOCK_DIV4; //use safe SPI rate of (16MHz / 4 = 4MHz)
208  delay(10); // settle time
209 
210  //test reading after data rate change
211  int MP3Clock = Mp3ReadRegister(SCI_CLOCKF);
212  if(MP3Clock != 0x6000) return 5;
213 
214  SetVolume(40, 40);
215  // one would think the following patch would over write the volume.
216  // But the SCI_VOL register space is not in the VSdsp's WRAM space.
217  // Note to keep an eye on it for future patches.
218 
219  if (VSLoadUserCode("patches.053")) return 6;
220 
221  delay(100); // just a good idea to let settle.
222 
223  return 0; // indicating all was good.
224 }
